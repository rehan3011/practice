import java.util.*;
import java.lang.*;
import java.io.*;

class Codechef
{
	public static void main (String[] args) throws java.lang.Exception
	{
		Scanner sc = new Scanner(System.in);
		
		int n = sc.nextInt();
		int earning;
		int tax;
    		for(int i=0; i<n; i++){
        		earning = sc.nextInt();
        		tax = sc.nextInt();
        		
        		System.out.println(earning - tax);
    		}
	}
}
